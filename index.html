async function findOrCreateDriveFile(){
setSyncStatus('syncing','A procurar ficheiro no Drive...');
try{
const r=await fetch(`https://www.googleapis.com/drive/v3/files?spaces=appDataFolder&q=name='${DRIVE_FILE}'&fields=files(id,name,modifiedTime)`,{headers:{Authorization:'Bearer '+accessToken}});
const data=await r.json();
if(data.files&&data.files.length>0){
driveFileId=data.files[0].id;
const mod=new Date(data.files[0].modifiedTime).toLocaleString('pt-PT');
setSyncStatus('connected',`Ligado · última atualização ${mod}`);
await loadFromDrive(true);
}else{await syncToDrive(true);}
}catch(e){setSyncStatus('error','Erro ao aceder ao Google Drive');}
}

async function syncToDrive(silent){
if(!accessToken){alert('Liga primeiro ao Google Drive.');return;}
setSyncStatus('syncing','A guardar no Drive...');
try{
const content=JSON.stringify(S,null,2);
if(driveFileId){
await fetch(`https://www.googleapis.com/upload/drive/v3/files/${driveFileId}?uploadType=media`,{method:'PATCH',headers:{Authorization:'Bearer '+accessToken,'Content-Type':'application/json'},body:content});
}else{
const meta=JSON.stringify({name:DRIVE_FILE,parents:['appDataFolder']});
const form=new FormData();
form.append('metadata',new Blob([meta],{type:'application/json'}));
form.append('file',new Blob([content],{type:'application/json'}));
const r=await fetch('https://www.googleapis.com/upload/drive/v3/files?uploadType=multipart&fields=id',{method:'POST',headers:{Authorization:'Bearer '+accessToken},body:form});
const d=await r.json();driveFileId=d.id;
}
const now=new Date().toLocaleString('pt-PT');
setSyncStatus('connected',`Guardado · ${now}`);
if(!silent)showToast('Dados guardados no Google Drive ✓');
}catch(e){setSyncStatus('error','Erro ao guardar');if(!silent)alert('Erro ao guardar no Google Drive.');}
}

async function loadFromDrive(silent){

setSyncStatus('syncing','A carregar do Drive...');
try{
const r=await fetch(`https://www.googleapis.com/drive/v3/files/${driveFileId}?alt=media`,{headers:{Authorization:'Bearer '+accessToken}});
const data=await r.json();S=data;saveState();
const now=new Date().toLocaleString('pt-PT');
setSyncStatus('connected',`Carregado · ${now}`);
nav(currentPage);
if(!silent)showToast('Dados carregados do Google Drive ✓');
}catch(e){setSyncStatus('error','Erro ao carregar do Drive');}
}
